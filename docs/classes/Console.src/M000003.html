  def setalpha(args)
    if args == "usage"
      put("setalpha [integer]: Sets the alpha of console to int")
    elsif args.length > 0
      if (0..255) === args[0].to_i
        @alpha = args[0].to_i
        put("Alpha set to #{@alpha}")
      else
        put("Alpha must be between 0 and 255")
      end
    end
  end
